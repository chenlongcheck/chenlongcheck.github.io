防抖和节流

2020-10-22 1 min read # 防抖和节流

HTML

<button onclick="lazyFunc()">防抖</button>
<button onclick="lazyFunc2()">节流</button>

JS

function fn() {
    console.log('防抖', this);
}
function debounce(func, wait = 1000) {
    // 当前的作用域是不销毁的
    let timer = null;
    return function () {
        // 当前作用域的this是btn元素
        clearTimeout(timer);
        timer = setTimeout(() => {
            func.call(this);
            timer = null;
        }, wait);
    }
}
let lazyFunc = debounce(fn, 1000);


function throttle(func, wait = 1000) {
    let timer = null;
    return function () {
        if (!timer)
            timer = setInterval(() => {
                func.call(this)
                clearInterval(timer)
                timer = null
            }, wait)
    }
}
function fn2() {
    console.log('节流', this)
}
let lazyFunc2 = throttle(fn2, 1000);
